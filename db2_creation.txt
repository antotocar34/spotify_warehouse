CREATE TABLE "dimension_artists" (
	"artist_uri"	TEXT NOT NULL UNIQUE,
	"artist_name"	TEXT,
	PRIMARY KEY("artist_uri")
)

CREATE TABLE "dimension_playlists" (
	"playlist_name"	TEXT,
	"collaborative"	TEXT,
	"num_tracks"	INTEGER,
	"num_artists"	INTEGER,
	"playlist_id"	INTEGER NOT NULL UNIQUE,
	PRIMARY KEY("playlist_id")
)

CREATE TABLE "dimension_songs" (
	"duration_ms"	REAL,
	"track_uri"	TEXT NOT NULL UNIQUE,
	"track_name"	TEXT,
	PRIMARY KEY("track_uri")
)

CREATE TABLE "dimension_top_artists" (
	"artist_uri"	TEXT NOT NULL UNIQUE,
	"artist_name"	TEXT,
	PRIMARY KEY("artist_uri")
)

CREATE TABLE "dimension_top_songs" (
	"track_uri"	TEXT NOT NULL UNIQUE,
	"track_name"	TEXT,
	"explicit"	TEXT,
	"duration_ms"	INTEGER,
	PRIMARY KEY("track_uri")
)

CREATE TABLE "fact_playlist" (
	"playlist_id"	INTEGER NOT NULL,
	"track_uri"	TEXT NOT NULL,
	"pos"	INTEGER NOT NULL,
	"danceability"	REAL,
	"energy"	REAL,
	"key"	REAL,
	"loudness"	REAL,
	"mode"	REAL,
	"speechiness"	REAL,
	"acousticness"	REAL,
	"instrumentalness"	REAL,
	"liveness"	REAL,
	"valence"	REAL,
	"tempo"	REAL,
	"type"	TEXT,
	"duration_ms"	REAL,
	"time_signature"	REAL,
	"collaborative"	TEXT,
	"num_tracks"	INTEGER,
	"num_followers"	INTEGER,
	"num_artists"	INTEGER,
	"artist_uri"	TEXT,
	FOREIGN KEY("playlist_id") REFERENCES "dimension_playlists"("playlist_id"),
	PRIMARY KEY("playlist_id","track_uri","pos")
)

CREATE TABLE "fact_top_track" (
	"track_uri"	TEXT NOT NULL UNIQUE,
	"artist_uri"	TEXT,
	"popularity"	INTEGER,
	"danceability"	REAL,
	"energy"	REAL,
	"key"	REAL,
	"loudness"	REAL,
	"mode"	REAL,
	"speechiness"	REAL,
	"acousticness"	REAL,
	"instrumentalness"	REAL,
	"liveness"	REAL,
	"valence"	REAL,
	"tempo"	REAL,
	"duration_ms"	REAL,
	FOREIGN KEY("track_uri") REFERENCES "dimension_top_songs"("track_uri"),
	PRIMARY KEY("track_uri")
)
